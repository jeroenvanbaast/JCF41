package huffman;

import java.util.Comparator;

/**
 *
 * @author Jeroen
 */
public class huffmanCompare implements Comparator<letter>{
    
    @Override
     public int compare(letter o1, letter o2) {
        if (o1.getFrequency() > o2.getFrequency()) {
            return 1;
        }
        else
        {
            return -1;
        }
    }
    
}
